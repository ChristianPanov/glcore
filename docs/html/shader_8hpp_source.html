345std::uint32_t shader_program::create_program() const
346{
347 const std::uint32_t program { glCreateProgram() };
348
349 std::vector<std::uint32_t> shader_ids;
350 shader_ids.reserve(m_shaders.size());
351
352 for (const auto& [type, src] : m_shaders)
353 shader_ids.push_back(compile(type, src));
354
355 for (const auto& id : shader_ids)
356 glAttachShader(program, id);
357 glLinkProgram(program);
358 glValidateProgram(program);
359
360 for (const auto& id : shader_ids)
361 glDetachShader(program, id);
362 for (const auto& id : shader_ids)
363 glDeleteShader(id);
364
365 return program;
366}

379std::vector<shader> shader_program::parse_shaders(const std::string& source) const
380{
381 std::vector<shader> shaders;
382 std::string_view type_token { "#type" };
383
384 std::size_t pos { source.find(type_token, 0) };
385 while (pos != std::string::npos) {
386 const std::size_t eol { source.find_first_of("\r\n", pos) };
387 const std::size_t begin { pos + type_token.size() + 1 };
388 const std::size_t next_line_pos { source.find_first_not_of("\r\n", eol) };
389
390 std::string_view type { source.substr(begin, eol - begin).data() };
391 pos = source.find(type_token, next_line_pos);
392 shaders.push_back({ string_to_shader_type(type), pos == std::string::npos ? source.substr(next_line_pos) : source.substr(next_line_pos, pos - next_line_pos) });
393 }
394
395 return shaders;
396}

403bool shader_program::is_valid(std::uint32_t id) const
404{
405 int is_compiled {};
406 glGetShaderiv(id, GL_COMPILE_STATUS, &is_compiled);
407
408 if (is_compiled == GL_FALSE) {
409 int max_length {};
410 glGetShaderiv(id, GL_INFO_LOG_LENGTH, &max_length);
411 std::vector<char> error_log(max_length);
412 glGetShaderInfoLog(id, max_length, &max_length, &error_log[0]);
413 std::fwrite(error_log.data(), error_log.size(), 1, stdout);
414 std::fwrite("\n", 1, 1, stdout);
415 glDeleteShader(id);
416 return false;
417 }
418
419 return true;
420}
