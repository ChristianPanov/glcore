161std::uint32_t shader_program::create_program() const
162{
163 const std::uint32_t program { glCreateProgram() };
164
165 std::vector<std::uint32_t> shader_ids;
166 shader_ids.reserve(m_shaders.size());
167
168 for (const auto& [type, src] : m_shaders)
169 shader_ids.push_back(compile(type, src));
170
171 for (const auto& id : shader_ids)
172 glAttachShader(program, id);
173 glLinkProgram(program);
174 glValidateProgram(program);
175
176 for (const auto& id : shader_ids)
177 glDetachShader(program, id);
178 for (const auto& id : shader_ids)
179 glDeleteShader(id);
180
181 return program;
182}

195std::vector<shader> shader_program::parse_shaders(const std::string& source) const
196{
197 std::vector<shader> shaders;
198 std::string_view type_token { "#type" };
199
200 std::size_t pos { source.find(type_token, 0) };
201 while (pos != std::string::npos) {
202 const std::size_t eol { source.find_first_of("\r\n", pos) };
203 const std::size_t begin { pos + type_token.size() + 1 };
204 const std::size_t next_line_pos { source.find_first_not_of("\r\n", eol) };
205
206 std::string_view type { source.substr(begin, eol - begin).data() };
207 pos = source.find(type_token, next_line_pos);
208 shaders.push_back({ string_to_shader_type(type), pos == std::string::npos ? source.substr(next_line_pos) : source.substr(next_line_pos, pos - next_line_pos) });
209 }
210
211 return shaders;
212}

219bool shader_program::is_valid(std::uint32_t id) const
220{
221 int is_compiled {};
222 glGetShaderiv(id, GL_COMPILE_STATUS, &is_compiled);
223
224 if (is_compiled == GL_FALSE) {
225 int max_length {};
226 glGetShaderiv(id, GL_INFO_LOG_LENGTH, &max_length);
227 std::vector<char> error_log(max_length);
228 glGetShaderInfoLog(id, max_length, &max_length, &error_log[0]);
229 std::fwrite(error_log.data(), error_log.size(), 1, stdout);
230 std::fwrite("\n", 1, 1, stdout);
231 glDeleteShader(id);
232 return false;
233 }
234
235 return true;
236}
